->How to move the audittable records to different tablespace when
the database is running

login as sys

Create the tablespace
------------------
CREATE TABLESPACE AUDIT_TRAIL_TS 
DATAFILE '/usr2/opt/oracle/oradata/gurudb/AUDDATA.dbf'
SIZE 500M
SEGMENT SPACE MANAGEMENT AUTO

MOVING AUD$ TO DIFFERENT TABLESPACE
---------------------------
begin
DBMS_AUDIT_MGMT.SET_AUDIT_TRAIL_LOCATION(
AUDIT_TRAIL_TYPE =>DBMS_AUDIT_MGMT.AUDIT_TRAIL_AUD_STD,
AUDIT_TRAIL_LCOATION_VALUE =>'AUDIT_TRAIL_TS'
);
END;

MOVING FINED GRAINED TO DIFFERENT TABLESPACE
-----------------------------
begin
DBMS_AUDIT_MGMT.SET_AUDIT_TRAIL_LOCATION(
AUDIT_TRAIL_TYPE =>DBMS_AUDIT_MGMT.AUDIT_TRAIL_FGA_STD,
AUDIT_TRAIL_LCOATION_VALUE =>'AUDIT_TRAIL_TS'
);
END;

MOVING BOTH AUD AND FINED GRAINED TO DIFFERENT TABLESPACE
--------------------------------

begin
DBMS_AUDIT_MGMT.SET_AUDIT_TRAIL_LOCATION(
AUDIT_TRAIL_TYPE =>DBMS_AUDIT_MGMT.AUDIT_TRAIL_DB_STD,
AUDIT_TRAIL_LCOATION_VALUE =>'AUDIT_TRAIL_TS'
);
END;

->pURGING OLD DATA

-------------------------------------
Auditing modifications of the data in the audit trail itself can be achieved as follows:

AUDIT INSERT, UPDATE, DELETE ON sys.aud$ BY ACCESS;

AUDIT DROP ANY TABLE BY SCOTT BY SESSION WHENEVER SUCCESSFUL;
