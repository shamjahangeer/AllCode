   AND MTD.ITEM_CATEGORY_ID = PP.ITEM_CATEGORY_ID 
   AND MTD.ITEM_CATEGORY_VERSION = CI1.ITEM_CATEGORY_VERSION 
   AND MTD.ITEM_CATEGORY_VERSION = PP.ITEM_CATEGORY_VERSION
